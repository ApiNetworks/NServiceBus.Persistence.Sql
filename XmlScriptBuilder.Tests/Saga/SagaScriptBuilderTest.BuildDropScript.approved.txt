
declare @tableName nvarchar(max) = '[' + @schema + '].[' + @endpointName + '.theSaga]';

IF EXISTS
(
    SELECT *
    FROM sys.objects
    WHERE
        object_id = OBJECT_ID(@tableName)
        AND type in (N'U')
)
BEGIN
    DECLARE @createTable nvarchar(max);
    SET @createTable = N'DROP TABLE ' + @tableName + '';
    exec(@createTable);
END
