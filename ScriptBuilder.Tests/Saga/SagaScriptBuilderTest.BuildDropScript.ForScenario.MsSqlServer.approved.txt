
/* TableNameVariable */ 

declare @tableName nvarchar(max) = @tablePrefix + 'theSaga';

/* DropTable */ 

if exists
(
    select *
    from sys.objects
    where
        object_id = object_id(@tableName)
        and type in ('U')
)
begin
    declare @createTable nvarchar(max);
    set @createTable = 'drop table  if exists ' + @tableName;
    exec(@createTable);
end
